//n ≥ 3
func fibbonacciDynamic(n int) []int {
	table := make([]int, n+1)
	table[0] = 0
	table[1] = 1
	for i := 2; i <= n; i++ {
    // 动态规划，子问题环环相扣构成母问题
		table[i] = table[i-1] + table[i-2] 
	}

	return table
}

func fibbonacci(n int) int {
	if n < 2 {
		return n
	}
	return fibbonacci(n-1) + fibbonacci(n-2)
}
